include(MacroMangosSourceGroup)

#-----------------------------------------------------------------------------
# Define the eluna library
file(GLOB sources *.cpp)
file(GLOB headers *.h)

set(eluna_LIB_SRCS ${sources} ${headers})

mangos_source_group(${eluna_LIB_SRCS})

include_directories(
    ${ACE_INCLUDE_DIRS}
    ${LUA_INCLUDE_DIR}
    ${MYSQL_INCLUDE_DIR}
    ${CMAKE_SOURCE_DIR}/dep/include/g3dlite/
    ${CMAKE_SOURCE_DIR}/dep/recastnavigation/Detour/Include/
    ${CMAKE_SOURCE_DIR}/src/framework/
    ${CMAKE_SOURCE_DIR}/src/shared/
    ${CMAKE_SOURCE_DIR}/src/game/
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${CMAKE_CURRENT_BINARY_DIR}
    ${CMAKE_BINARY_DIR}
    ${CMAKE_BINARY_DIR}/src/shared/
)

#-----------------------------------------------------------------------------
# Build the eluna library
add_library(mangos-eluna STATIC ${eluna_LIB_SRCS})
target_link_libraries(mangos-eluna mangos-shared mangos-framework Detour g3dlite ${ACE_LIBRARIES} ${LUA_LIBRARIES} ${MYSQL_LIBRARIES})

if(${CMAKE_BUILD_TYPE} MATCHES Debug)
    set(BUILD_PROPERTIES "-DMANGOS_DEBUG")
    set_target_properties(mangos-eluna PROPERTIES COMPILE_FLAGS ${BUILD_PROPERTIES})
endif()
